
принт('мен обедге кетип жатам')
money = инпут('мага акча берчи обед кылганы: ')

эгерде money >= 1000:
    принт('Кут Ордого барып жирный обед кылым келем.')

эгерде money < 1000 жана money >= 600:
    принт('Царский дворго барып шашлык жеп келейинчи.')

эгерде money < 600 жана money >= 400:
    принт('Лагманныйга барып лагман жеп келем.')

эгерде money > 200 жана money < 400:
    принт('Хмм 1 порций аш жейт окшойм.')

эгерде money <= 200:
    принт('Уйго барып обед кылып келет окшойм.')
